🗄️ Database Schema (PostgreSQL)
CREATE TABLE users ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), name VARCHAR(120) NOT NULL, email VARCHAR(255) UNIQUE NOT NULL, password VARCHAR(255) NOT NULL, university VARCHAR(200), role ENUM('student','admin') DEFAULT 'student', created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE TABLE projects ( id UUID PRIMARY KEY DEFAULT gen_random_uuid(), title VARCHAR(255) NOT NULL, description TEXT NOT NULL, category VARCHAR(80), status ENUM('open','progress','closed') DEFAULT 'open', tech_stack TEXT[], author_id UUID REFERENCES users(id) ON DELETE CASCADE, open_collab BOOLEAN DEFAULT true, likes_count INT DEFAULT 0, created_at TIMESTAMPTZ DEFAULT NOW() ); CREATE INDEX idx_projects_category ON projects(category); CREATE INDEX idx_projects_status ON projects(status); CREATE INDEX idx_projects_created ON projects(created_at DESC);
